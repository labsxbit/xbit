class Student{

 }

Using Methods

Using methods to find even or odd number Dec 19, 2022


import java.util.Scanner;

public class Main{
    public static int even_odd(int num){
		if(num%2==0){
			System.out.println("Even number\n");
			return 1;
		}
		else{
			System.out.println("Odd number\n");		
			return 0;
		}
	}
	
	public static void main(String[] args){
		int num;
		// here we create an object called scan which is created from Scanner class 
		Scanner scan = new Scanner(System.in);
		
		System.out.println("Enter a number: ");
		// the same object we created is now used to call nextInt() method which assumes the user is entering an integer input value. 
		num = scan.nextInt();
		// here we use an if condition to check if the remainder when divided by 2 gives 0 then the number will be even otherwise odd. 
		
		even_odd(num);
	}
}

// Expected output
Enter a number: 
123
Odd number
